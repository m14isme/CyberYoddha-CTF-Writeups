for x, y in zip(orig.split(' '), task.split(' ')):
    if x == y:
        continue
    i = 0
    try:
        while x[i] == y[i]:
            i += 1
    except:
        pass
    flag += x[i] if len(x) > len(y) else y[i]

print(f'[+] Reconstructed flag: "{flag}" ... ')
